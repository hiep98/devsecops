Set static ip

sudo vi /etc/netplan/01-netcfg.yaml

network:

    version: 2

    renderer: networkd

    ethernets:

        ens33:

            addresses:

                - 192.168.207.129/24

            nameservers:

                addresses: [8.8.8.8, 8.8.4.4]

            routes:

                - to: default

                  via: 192.168.207.2

 

sudo netplan apply

sudo ip addr del 192.168.207.131/24 dev ens33
